# @xfe/common-login专题通用登录弹窗改版

web端专题通用登录

# 浏览器

IE9 +

# 浏览器使用

# 正式环境

在浏览器中, 开发者可以直接通过引用的方式使用, 其中 0.0.x 为当前 changeLog 最新版本, 如需指定版本请直接更换即可, 如:

<link rel="stylesheet" type="text/css" href="//zhcdn01.xoyo.com/xassets/lib/common-login/0.0.x/master/xpass.umd.css" />
<script src="//zhcdn01.xoyo.com/xassets/lib/common-login/0.0.x/master/xpass.umd.js"></script>

# 测试环境

在浏览器中, 开发者可以直接通过引用的方式使用, 其中 0.0.x 为当前 changeLog 最新版本, 如需指定版本请直接更换即可, 如:

<link rel="stylesheet" type="text/css" href="//zhcdn01.xoyo.com/xassets/lib/common-login/0.0.x/test/xpass.umd.css" />

# Get Started

# 基本用法

# 登录

    /**
     * 点击登录
     * @param props 例如:props={onLoginSuccess:()=>{console.log('登录成功回调')},source:101}
     */
window.XPASS.signin(props);

# 注册

    /**
     * 点击注册
     * @param props 例如:props={onRegisterSuccess:()=>{console.log('注册成功回调')},source:202}
     */
window.XPASS.signup(props);

# 退出

window.XPASS.exit();

# 退出金山通行证

window.XPASS.exitPassport();

# 退出第三方登录(QQ,微信,微博)

window.XPASS.exitThird();

# 获取登录账号信息

//获取登录账号信息
const accountInfo = window.XPASS.u_account;
/**
 * 因为允许同时存在第三方登录和金山通行证登录状态，所以为方便业务灵活控制需要显示的账号信息，添加多xoyo_account和third_account两个字段
 */
//获取金山通行证登录账号信息
const accountInfo = window.XPASS.xoyo_account;
//获取第三方登录账号信息
const accountInfo = window.XPASS.third_account;
